Example 5 (dynamic data source):

Run "node server.js" in "Examples" folder to see results

// Creating InfiniteAutocomplete instance by passing the Root Element in the first argument
// with passing the "getDataFromApi" configuration to fetch dynamic data from Server EndPoint.

var test5 = new InfiniteAutocomplete(
  document.getElementById('test5'),
  {
    onSelect: function(targetElement, selectedOption) {
      console.log("selectedOption", selectedOption);
    },
    getDataFromApi: function(text, page, fetchSize) {
        return new Promise(function(resolve, reject) {
            function reqListener () {
                resolve(JSON.parse(this.responseText));
            }

            var oReq = new XMLHttpRequest();
            oReq.addEventListener("load", reqListener);
            oReq.open("GET", "http://localhost:5000/data?text="+ text + "&page=" + page + "&fetchSize=" + fetchSize);
            oReq.send();
        });
    }
  }
);

Advanced Example:

var CustomizedInputForAdvancedTest = (function () { // ES5 Class
  function CustomizedInputForAdvancedTest() {
  }
  CustomizedInputForAdvancedTest.prototype.render = function () {
      return '< div>Start typing!< input type="text" style="background:red;"/>';
  };
  CustomizedInputForAdvancedTest.prototype.onInputChange = function (inputDOMElement, value) {
      console.log(value);
  };
  return CustomizedInputForAdvancedTest;
}());


var OptionsComponentForAdvancedTest = (function () { // ES5 Class
  function OptionsComponentForAdvancedTest() {
      this.listElementSelector = ".myList";
  }
  OptionsComponentForAdvancedTest.prototype.render = function () {
      //You can inject template code before
      return "< div>Options:< div class='myList'>";
      //You can inject template code after
  };
  OptionsComponentForAdvancedTest.prototype.renderOption = function (option) {
      return "< li style='font-weight:bold;'>" + option.text + "";
  };
  return OptionsComponentForAdvancedTest;
}());

var advancedTest = new InfiniteAutocomplete(
  document.getElementById('advanced-test'),
  {
      customizedInput: CustomizedInputForAdvancedTest,
      customizedOptions: OptionsComponentForAdvancedTest
  }
);

advancedTest.setConfig({data:[
  { text: 'first', value: 1},
  { text: 'second', value: 2},
  { text: 'theird', value: 3},
  { text: 'fourth', value: 4},
  { text: 'fivth', value: 5}
]});
